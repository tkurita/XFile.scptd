XFile

What ?

XFile is a module of AppleScript to provides unified object oriented interface to file operation (moving, removing, getting attributes and so on).

Functions of file operations of AppleScript world is given by Scripting Additions, Finder, System Events and shell commands. Functions are distributed and cumbersome. XFile wraps these components and introduces a file object for unified object oriented interface.

Sample

property XFile : module

boot (module loader) for me

(*== Make a New Instance *)
set a_xfile to XFile's make_with(alias "Macintosh HD:Users:")
set a_xfile to XFile's make_with("/Users")

(*== Obtain File Infomation *)
log a_xfile's is_folder() -- true
log a_xfile's is_package() -- false
log a_xfile's item_name() -- "Users"

(*== Obtain Parent and Child *)
log a_xfile's parent_folder()'s hfs_path() -- "Macintosh HD:"
log a_xfile's child("tkurita")'s hfs_path() -- "Macintosh HD:Users:tkurita"
log a_xfile's child("tkurita:Documents")'s hfs_path()
-- "Macintosh HD:Users:tkurita:Documents"
log a_xfile's unique_child("tkurita")'s hfs_path()
--"Macintosh HD:Users:tkurita 2"

(*== Read and Write *)
set home_folder to XFile's make_with(path to home folder)
set test_file to home_folder's child("testfile")
test_file's write_as_utf8("new data")
log test_file's read_as_utf8() -- "new data"

(*== File Manipulations *)
log test_file's item_exists() --true

set test_file2 to test_file's copy_to(home_folder's child("testfile2"))
log test_file2's hfs_path() --Macintosh HD:Users:tkurita:testfile2

test_file2's move_to(home_folder's child("Documents"))
log test_file2's hfs_path() -- "Macintosh HD:Users:tkurita:Documents:testfile2"

test_file2's into_trash()
log test_file2's hfs_path() -- "Macintosh HD:Users:tkurita:.Trash:testfile2"
test_file's into_trash()
